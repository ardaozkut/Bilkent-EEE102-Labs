library IEEE;
use IEEE.STD_LOGIC_1164.ALL;

entity source1 is
    Port ( x1 : in STD_LOGIC;
           x2 : in STD_LOGIC;
           x3 : in STD_LOGIC;
           x4 : in STD_LOGIC;
           x5 : in STD_LOGIC;
           x6 : in STD_LOGIC;
           inout1 : inout std_logic;
           out1 : out STD_LOGIC);
end source1;

architecture Behavioral of source1 is

begin
inout1 <= (x1 and x2 and not x3);
out1 <= ((not inout1) or (inout1 and x5)) and not x6;

end Behavioral;
